diagnostic.sh — فحص شامل عبر ADB 
#!/bin/bash
# Phone Diagnostic Script - Comprehensive Check

echo "========================================"
echo "  📱 Phone Diagnostic Report"
echo "========================================"
echo ""

# 1. Device Info
echo "[1] Device Information:"
echo "   Model: $(adb shell getprop ro.product.model | tr -d '\r')"
echo "   Android: $(adb shell getprop ro.build.version.release | tr -d '\r')"
echo "   Security Patch: $(adb shell getprop ro.build.version.security_patch | tr -d '\r')"
echo "   Build: $(adb shell getprop ro.build.display.id | tr -d '\r')"
echo ""

# 2. Battery Status
echo "[2] Battery Status:"
adb shell dumpsys battery | grep -E "level|status|health|temp|voltage" | sed 's/^/   /'
echo ""

# 3. Storage
echo "[3] Storage Status:"
adb shell df -h /data | tail -1 | awk '{print "   Total: " $2, "Used: " $3, "Free: " $4, "Usage: " $5}'
echo ""

# 4. Network
echo "[4] Network Status:"
echo "   Baseband: $(adb shell getprop gsm.version.baseband | tr -d '\r')"
echo "   IMEI: $(adb shell service call iphonesubinfo 1 | grep -o '[0-9]*' | tr -d '\n')"
adb shell dumpsys telephony.registry | grep -E "mServiceState|mSignalStrength" | head -2 | sed 's/^/   /'
echo ""

# 5. Boot Status
echo "[5] Boot Status:"
echo "   Bootloader: $(adb shell getprop ro.bootloader | tr -d '\r')"
echo "   Boot Reason: $(adb shell getprop ro.boot.bootreason | tr -d '\r')"
echo "   Verified Boot: $(adb shell getprop ro.boot.verifiedbootstate | tr -d '\r')"
echo ""

# 6. Running Services
echo "[6] Critical Services:"
adb shell dumpsys activity services | grep -E "com.android.phone|com.android.systemui|android.hardware" | head -5 | sed 's/^/   /'
echo ""

echo "[✓] Diagnostic Complete!"

🎯 تشخيص حسب المشكلة

المشكلة: الجهاز لا يُشحن

🔋 لا يُشحن
⬇
هل الكابل/الشاحن يعمل مع جهاز آخر؟
❌ لا → غيّر الشاحن✅ نعم → ⬇
⬇
هل منفذ USB نظيف؟
❌ لا → نظف بالفرشاة✅ نعم → ⬇
⬇
هل البطارية منتفخة؟
✅ نعم → استبدل البطارية فوراً!❌ لا → ⬇
⬇
فحص IC الشحن بالمتعدد → استبدل IC إذا لزم

المشكلة: لا شبكة / IMEI مفقود

📡 لا شبكة
⬇
هل Baseband معروف؟ (*#06#)
❌ Unknown → ⬇✅ معروف → ⬇
⬇
❌ فلّش Modem/CP مناسب + استعادة EFS
⬇
✅ فحص إعدادات APN + Network Mode + SIM

المشكلة: شاشة سوداء / لا تعمل

🖤 شاشة سوداء
⬇
هل الجهاز يهتز/يصدر صوت؟
✅ نعم → الشاشة/كابل LCD تالف❌ لا → ⬇
⬇
هل يدخل Fastboot؟
✅ نعم → مشكلة Software❌ لا → مشكلة PMIC/هاردوير
