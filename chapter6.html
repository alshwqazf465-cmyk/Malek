🔬 فحص IC الشحن بالمتعدد

فحص اللوحة الأم بالمتعدد — التحقق من جهد VBUS و VBAT
نقطة الفحص الجهد الطبيعي إذا كان 0V
VBUS (منفذ USB) 5.0V عند الشحن كابل/شاحن تالف، منفذ USB مكسور
VBAT (بطارية) 3.7-4.2V بطارية ميتة، IC شحن تالف
PMIC VCC 1.8V, 3.3V PMIC تالف → الجهاز ميت
VBUCK (CPU) 0.8-1.2V PMIC تالف → Bootloop

📱 سكريبت مراقبة البطارية

battery-monitor.sh — مراقبة مستمرة 
#!/bin/bash
# Battery Monitor - Logs battery stats over time

LOGFILE="battery_log_$(date +%Y%m%d).csv"
echo "timestamp,level,status,health,voltage,temp,current" > "$LOGFILE"

echo "[*] Monitoring battery... Press Ctrl+C to stop"

while true; do
    LEVEL=$(adb shell dumpsys battery | grep level | awk '{print $2}')
    STATUS=$(adb shell dumpsys battery | grep status | head -1 | awk '{print $2}')
    HEALTH=$(adb shell dumpsys battery | grep health | head -1 | awk '{print $2}')
    VOLT=$(adb shell dumpsys battery | grep voltage | awk '{print $2}')
    TEMP=$(adb shell dumpsys battery | grep temperature | awk '{print $2}')
    CURR=$(adb shell dumpsys battery | grep 'Max charging current' | awk '{print $4}')
    
    echo "$(date '+%H:%M:%S'),$LEVEL,$STATUS,$HEALTH,$VOLT,$TEMP,$CURR" >> "$LOGFILE"
    
    # Alert if temperature > 45°C (450 in dumpsys)
    if [ "$TEMP" -gt 450 ]; then
        echo "[!] WARNING: Battery temperature > 45°C!"
        adb shell input keyevent 26  # Turn off screen
    fi
    
    sleep 60
done
